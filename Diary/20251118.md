	#define BSP_ADKEY_USE_INNER_PU_10K  0               // 如果外部没有上拉电阻时，将这里配置为1
#define BSP_ADKEY_PULLUP_RESISTANCE (100*1000)	    // 外部使用的上拉电阻阻值，单位(欧)
#define BSP_ADKEY_VALUE_RANGE       (10)            // 键值范围，如果电阻太接近，需要减小这个值

// 定义键值，从1开始
#define USER_KEY_POWER   1
#define USER_KEY_OK      2
#define USER_KEY_DOWN    3
#define USER_KEY_UP      4
#define USER_KEY_MENU    5
#define USER_KEY_MODE    6

// 以下定义 ADKEY 电阻和键值之间的映射关系，按电阻值从小到大排序
#define BSP_ADKEY_CODE_MAP \
    ADKEY_CODE(3*1000,   USER_KEY_DOWN), \
    ADKEY_CODE(6.2*1000, USER_KEY_OK), \
    ADKEY_CODE(10*1000,  USER_KEY_MENU), \
    ADKEY_CODE(15*1000,  USER_KEY_UP), \
    ADKEY_CODE(390*1000, USER_KEY_MODE), \

	#define ADKEY_PU_RES() (BSP_ADKEY_USE_INNER_PU_10K ? \
                        (10000 + BSP_ADKEY_PULLUP_RESISTANCE) : BSP_ADKEY_PULLUP_RESISTANCE)
#define ADKEY_VALUE(key_res) (((key_res) * (1 << ADC_RESOLUTION_BITS)) / ((key_res) + ADKEY_PU_RES()))
#define ADKEY_CODE(key_res, code) {my_min(ADKEY_VALUE((u32)key_res) + BSP_ADKEY_VALUE_RANGE, (1 << ADC_RESOLUTION_BITS)), code}
}
